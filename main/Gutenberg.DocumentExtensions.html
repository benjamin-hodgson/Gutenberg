Remarks

docs.SeparatedAndTerminated(sep) is equivalent to documents.Separated(separator).Append(separator).

Examples
var doc = new Doc[] { "item1", "item2", "item3" }
    .SeparatedAndTerminated(",");
Console.WriteLine(doc);
// Output:
// item1,item2,item3,
